    function App() {
      const [page, setPage] = React.useState('home');
      const [theme, setTheme] = React.useState('light');
      const [lang, setLang] = React.useState('IT');
      const [anim, setAnim] = React.useState('piene');

      React.useEffect(() => {
        const h = document.documentElement;
        h.dataset.theme = theme; h.dataset.anim = anim;
        h.lang = lang === 'UK' ? 'uk' : lang.toLowerCase();

        home: window.DashboardScreen, portafoglio: window.PortfolioScreen, pac: window.PacScreen,
