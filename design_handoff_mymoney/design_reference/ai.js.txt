/* MyMoney — real AI bridge (Gemini). Key lives only in this browser (localStorage). */
window.MMAI = {
  getKey() { try { return localStorage.getItem('mm_ai_key') || ''; } catch (e) { return ''; } },
  setKey(k) { try { localStorage.setItem('mm_ai_key', k || ''); } catch (e) {} },
  getModel() { try { return localStorage.getItem('mm_ai_model') || 'gemini-2.0-flash'; } catch (e) { return 'gemini-2.0-flash'; } },
  setModel(m) { try { localStorage.setItem('mm_ai_model', m || 'gemini-2.0-flash'); } catch (e) {} },
  enabled() { return !!this.getKey().trim(); },

  async complete(prompt) {
    const key = this.getKey().trim();
    if (!key) { const e = new Error('Nessuna chiave API'); e.code = 'no-key'; throw e; }
    const model = this.getModel().trim() || 'gemini-2.0-flash';
    const url = 'https://generativelanguage.googleapis.com/v1beta/models/' + encodeURIComponent(model) + ':generateContent?key=' + encodeURIComponent(key);
    const res = await fetch(url, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ contents: [{ parts: [{ text: prompt }] }] }),
    });
    if (!res.ok) {
      const e = new Error('HTTP ' + res.status);
      e.status = res.status;
      throw e;
    }
    const j = await res.json();
    const parts = (((j.candidates || [])[0] || {}).content || {}).parts || [];
    return parts.map((p) => p.text || '').join('').trim();
  },

  async completeJSON(prompt) {
    const raw = await this.complete(prompt + '\n\nRispondi SOLO con JSON valido, senza markdown né testo extra.');
    const m = raw.match(/\{[\s\S]*\}/);
    if (!m) throw new Error('Risposta non in formato JSON');
    return JSON.parse(m[0]);
  },

  async test() {
    const t = await this.complete('Rispondi con una sola parola: ok');
    return /ok/i.test(t) || t.length > 0;
  },

  portfolioContext() {
    const D = window.MM_DATA || {};
    const rows = ((D.portfolio || {}).rows || []).map((r) => r.tk + ' (' + r.name + ', ' + r.cat + ', valore €' + r.valore + ', P/L ' + r.pl + '%)').join('; ');
    return 'Contesto portafoglio (dati privati dell\u2019utente, app MyMoney): patrimonio totale €' + (D.patrimonio || 0) +
      ', performance 12 mesi ' + (D.perf12m || 0) + '%. Posizioni: ' + rows + '.';
  },
};
