/* MyMoney — Analisi · PAC · Notizie · Impostazioni. */
(() => {
  const MM = window[Object.keys(window).find((k) => /^MyMoneyDesignSystem/.test(k))];
  const { StatCard, Card, Pill, Badge, Button, DataTable, Note, FormField, Input, Select } = MM;
  const F = window.MMFmt;
  const Icon = window.MMIcon;

  /* ---------------- Analisi ---------------- */
  function AnalisiScreen() {
    const a = window.MM_DATA.analisi;
    return (
      <div>
        <div className="page-title">Analisi <span className="ai-label" style={{ verticalAlign: 'middle' }}>look-through</span></div>
        <p className="page-sub">Esposizione reale e rischio del portafoglio, scomponendo gli ETF.</p>
        <div className="grid">
          <StatCard tone={a.techAlert ? 'warn' : 'ok'} label="Concentrazione tecnologia" value={F.pct(a.techConc, 0)} hint={a.techAlert ? 'Sopra soglia 25%' : 'Entro soglia'} />
          <StatCard label="Dividend yield medio" value={F.pct(a.divYield)} />
          <StatCard label="Diversificazione" value={F.num(a.effHoldings)} hint="titoli effettivi (look-through)" />
        </div>
        <Card variant="glass" eyebrow="Esposizione settoriale" style={{ marginTop: '18px' }}>
          {a.settori.map((s, i) => (
            <div key={s.name} style={{ margin: '8px 0' }}>
              <div style={{ display: 'flex', justifyContent: 'space-between', fontSize: '13px', marginBottom: '3px' }}><span>{s.name}</span><span className="muted num">{F.pct(s.pct, 0)}</span></div>
              <div style={{ background: 'var(--surface-alt)', borderRadius: '999px', height: '9px', overflow: 'hidden' }}>
                <div style={{ height: '100%', width: s.pct + '%', background: i === 0 ? 'var(--accent)' : 'var(--accent-strong)', borderRadius: '999px', opacity: 1 - i * 0.08 }} />
              </div>
            </div>
          ))}
          <p className="faint" style={{ fontSize: '12px', marginTop: '12px' }}>Stima su dati pubblici degli ETF. Non è un consiglio finanziario.</p>
        </Card>
        <Card eyebrow="Metriche di rischio" style={{ marginTop: '16px' }}>
          <div className="grid" style={{ gridTemplateColumns: 'repeat(auto-fit,minmax(150px,1fr))' }}>
            <StatCard label="Volatilità" value={F.pct(a.risk.vol)} hint="annua" style={{ boxShadow: 'none', border: '1px solid var(--border)' }} />
            <StatCard tone="warn" label="Max drawdown" value={F.signedPct(a.risk.mdd)} hint="peggior calo" style={{ boxShadow: 'none', border: '1px solid var(--border)' }} />
            <StatCard label="Sharpe" value={F.num(a.risk.sharpe, 2)} hint="rendimento/rischio" style={{ boxShadow: 'none', border: '1px solid var(--border)' }} />
            <StatCard label="Beta" value={F.num(a.risk.beta, 2)} hint="vs. mercato" style={{ boxShadow: 'none', border: '1px solid var(--border)' }} />
          </div>
          <p className="faint" style={{ fontSize: '12px', marginTop: '10px' }}>Su {a.risk.weeks} settimane · aggiornato {a.risk.when}.</p>
        </Card>
      </div>
    );
  }
  window.AnalisiScreen = AnalisiScreen;

  /* ---------------- PAC ---------------- */
  function PacScreen() {
    const p = window.MM_DATA.pac;
    const [importo, setImporto] = React.useState(String(p.importo));
    const base = Number(importo) || 0;
    const scale = base / p.importo;
    const cols = [
      { key: 'tk', label: 'Titolo', render: (r) => <span><span className="t-ticker">{r.tk}</span> <span className="t-type">· {r.tipo}</span><br /><span className="muted" style={{ fontSize: '13px' }}>{r.name}</span></span> },
      { key: 'cat', label: 'Categoria', render: (r) => <span className="muted">{r.cat}</span> },
      { key: 'target', label: '% Target', align: 'right', render: (r) => F.pct(r.target) },
      { key: 'quota', label: 'Quota mensile', align: 'right', render: (r) => F.eur(r.quota * scale, 0) },
    ];
    return (
      <div>
        <div style={{ marginBottom: '18px' }}>
          <div className="page-title" style={{ margin: 0, marginBottom: '12px' }}>PAC — Piano di Accumulo</div>
          <p className="page-sub" style={{ margin: 0, marginBottom: '16px' }}>Inserisci quanto investire ogni mese: ripartiamo sui target.</p>
          <div style={{ display: 'grid', gridTemplateColumns: 'repeat(3, 1fr)', gap: '12px', marginBottom: '18px' }}>
            <div style={{ padding: '16px', background: 'var(--surface)', border: '1px solid var(--border)', borderRadius: 'var(--r-lg)', display: 'flex', flexDirection: 'column', justifyContent: 'center' }}>
              <div className="muted" style={{ fontSize: '11px', textTransform: 'uppercase', letterSpacing: 'var(--ls-caps)', fontWeight: 600, marginBottom: '6px' }}>Importo mensile</div>
              <div style={{ display: 'flex', gap: '10px', marginBottom: '8px' }}>
                <Input value={importo} onChange={(e) => setImporto(e.target.value)} placeholder="es. 1.000" />
                <Button variant="primary" onClick={(e) => { e.preventDefault(); }} style={{ whiteSpace: 'nowrap' }}>Calcola</Button>
              </div>
              <div className="muted" style={{ fontSize: '11px', lineHeight: 1.3 }}>La rata totale del piano di accumulo.</div>
            </div>
            <div style={{ padding: '16px', background: 'var(--surface)', border: '1px solid var(--border)', borderRadius: 'var(--r-lg)', display: 'flex', flexDirection: 'column', justifyContent: 'center' }}>
              <div className="muted" style={{ fontSize: '11px', textTransform: 'uppercase', letterSpacing: 'var(--ls-caps)', fontWeight: 600, marginBottom: '6px' }}>Importi fissi</div>
              <div className="num" style={{ fontSize: 'var(--text-lg)', fontWeight: 800, letterSpacing: 'var(--ls-tight)', color: 'var(--ink)', marginBottom: '8px' }}>{F.eur(p.sommaFissi)}</div>
              <div style={{ fontSize: '12px', lineHeight: 1.4, color: 'var(--muted)' }}>
                {p.fisse.map((f, i) => (
                  <div key={i}><span className="t-ticker">{f.tk}</span> · {F.eur(f.importo)}</div>
                ))}
                <div style={{ marginTop: '6px', fontSize: '11px', fontWeight: 600, color: 'var(--ink)' }}>{p.fisse.length} titolo{p.fisse.length > 1 ? 'i' : ''}</div>
              </div>
            </div>
            <div style={{ padding: '16px', background: 'var(--surface)', border: '1px solid var(--border)', borderRadius: 'var(--r-lg)', display: 'flex', flexDirection: 'column', justifyContent: 'center' }}>
              <div className="muted" style={{ fontSize: '11px', textTransform: 'uppercase', letterSpacing: 'var(--ls-caps)', fontWeight: 600, marginBottom: '6px' }}>Titoli</div>
              <div className="num" style={{ fontSize: 'var(--text-lg)', fontWeight: 800, letterSpacing: 'var(--ls-tight)', color: 'var(--ink)' }}>{p.rows.length}</div>
            </div>
          </div>
        </div>
        <DataTable columns={cols} rows={p.rows} getRowKey={(r) => r.tk}
          footer={[<b>Totale</b>, '', <b className="pos">{F.pct(p.sommaPct, 0)}</b>, <b>{F.eur(base)}</b>]} />
      </div>
    );
  }
  window.PacScreen = PacScreen;

  /* ---------------- Notizie ---------------- */
  function NotizieScreen() {
    const n = window.MM_DATA.notizie;
    const arrow = (d) => d === 'up' ? <Icon name="up" size={14} /> : d === 'down' ? <Icon name="down" size={14} /> : <span style={{ opacity: .6 }}>→</span>;
    const relTone = (r) => r >= 70 ? 'high' : r >= 50 ? 'mid' : 'low';
    return (
      <div>
        <div className="page-title">Notizie</div>
        <p className="page-sub">Dal monitor del tuo portafoglio · <span className="muted">aggiornato {n.aggiornato}</span></p>
        {n.cards.map((c, i) => (
          <Card key={i} style={{ marginBottom: '14px' }}>
            <div style={{ display: 'flex', alignItems: 'center', gap: '8px' }}>
              <span className="card-h" style={{ margin: 0, color: 'var(--muted)' }}>{c.tipo}</span>
              <Badge level={relTone(c.rilevanza)}>{c.rilevanza}/100</Badge>
            </div>
            <div className="card-title" style={{ marginTop: '6px' }}>{c.ticker ? <span className="t-ticker">{c.ticker}</span> : null} {c.ticker ? '— ' : ''}{c.titolo}</div>
            <div style={{ display: 'flex', gap: '6px', flexWrap: 'wrap', margin: '8px 0' }}>
              {c.impatti.map((p, j) => (
                <span key={j} className="pill" style={{ color: p.dir === 'up' ? 'var(--pos)' : p.dir === 'down' ? 'var(--neg)' : 'var(--muted)', background: p.dir === 'up' ? 'var(--pos-bg)' : p.dir === 'down' ? 'var(--neg-bg)' : 'var(--neutral-bg)' }}>
                  {p.hk} {arrow(p.dir)} {p.vw}
                </span>
              ))}
              <span className="pill gray">confidenza: {c.conf}</span>
            </div>
            <div className="muted" style={{ fontSize: '13px' }}>{c.data} · <a href="#">{c.fonte}</a></div>
          </Card>
        ))}
        <p className="faint" style={{ fontSize: '12px' }}>Sola lettura. Le notizie aiutano a capire, non danno segnali operativi.</p>
      </div>
    );
  }
  window.NotizieScreen = NotizieScreen;

  /* ---------------- Impostazioni ---------------- */
  function ImpostazioniScreen({ theme, setTheme, lang, setLang, anim, setAnim }) {
    const [saved, setSaved] = React.useState(false);
    const [aiTest, setAiTest] = React.useState(null);
    const [testing, setTesting] = React.useState(false);
    const [apiKey, setApiKey] = React.useState(window.MMAI.getKey());
    const [model, setModel] = React.useState(window.MMAI.getModel());
    const aiOn = !!apiKey.trim();
    const saveAI = () => {
      window.MMAI.setKey(apiKey.trim());
      window.MMAI.setModel(model.trim() || 'gemini-2.0-flash');
      setSaved(true); setTimeout(() => setSaved(false), 2200);
    };
    const testAI = async () => {
      window.MMAI.setKey(apiKey.trim());
      window.MMAI.setModel(model.trim() || 'gemini-2.0-flash');
      setTesting(true); setAiTest(null);
      try { await window.MMAI.test(); setAiTest('ok'); }
      catch (err) { setAiTest(err.status === 429 ? 'rate' : 'err'); }
      finally { setTesting(false); }
    };
    return (
      <div>
        <div className="page-title">Impostazioni</div>
        <p className="page-sub">Aspetto, lingua e agente AI.</p>
        {saved && <Note tone="ok">Preferenze salvate.</Note>}
        <Card variant="glass" eyebrow="Aspetto e lingua">
          <div className="row2" style={{ marginTop: '4px' }}>
            <FormField label="Tema"><Select value={theme} onChange={(e) => setTheme(e.target.value)} options={[{ value: 'light', label: 'Chiaro' }, { value: 'dark', label: 'Scuro' }]} /></FormField>
            <FormField label="Lingua"><Select value={lang} onChange={(e) => setLang(e.target.value)} options={['IT', 'EN', 'ES', 'FR', 'DE', 'UK']} /></FormField>
          </div>
          <FormField label="Animazioni" hint="Quanto vive l'interfaccia." style={{ maxWidth: '280px', marginBottom: 0 }}>
            <Select value={anim} onChange={(e) => setAnim(e.target.value)} options={[{ value: 'piene', label: 'Piene' }, { value: 'leggere', label: 'Leggere' }, { value: 'spente', label: 'Spente' }]} />
          </FormField>
        </Card>
        <Note tone="info">Le chiavi e i dati personali restano solo sul tuo computer, mai online.</Note>
        <Card eyebrow="Agente AI" title={<span style={{ display: 'inline-flex', alignItems: 'center', gap: '6px' }}><Icon name="sparkles" size={15} /> Gemini</span>} style={{ marginTop: '4px' }}>
          {aiOn
            ? <Note tone="ok" style={{ marginBottom: '12px' }}>Agente attivo · {model || 'gemini-2.0-flash'} — le funzioni AI dell'app usano la tua chiave.</Note>
            : <Note tone="warn" style={{ marginBottom: '12px' }}>Inserisci la chiave API per attivare l'agente. Senza chiave l'app usa risposte dimostrative.</Note>}
          {aiTest === 'ok' && <Note tone="ok">Connessione riuscita — l'agente risponde.</Note>}
          {aiTest === 'rate' && <Note tone="warn">Limite richieste (429). Riprova tra un minuto.</Note>}
          {aiTest === 'err' && <Note tone="warn">Connessione fallita: controlla chiave e modello.</Note>}
          <FormField label="Chiave API" hint="Resta solo su questo browser, mai inviata altrove.">
            <Input type="password" value={apiKey} onChange={(e) => setApiKey(e.target.value)} placeholder="AIza…" autoComplete="off" />
          </FormField>
          <div className="row2">
            <FormField label="Modello"><Input value={model} onChange={(e) => setModel(e.target.value)} placeholder="gemini-2.0-flash" /></FormField>
            <FormField label="Stato"><Input readOnly value={aiOn ? 'Attivo' : 'Disattivo — manca la chiave'} /></FormField>
          </div>
          <div className="actions" style={{ marginBottom: 0 }}>
            <Button variant="primary" onClick={saveAI}>Salva</Button>
            <Button variant="ghost" disabled={!aiOn || testing} onClick={testAI}>{testing ? 'Verifica…' : 'Prova connessione'}</Button>
          </div>
        </Card>
      </div>
    );
  }
  window.ImpostazioniScreen = ImpostazioniScreen;
})();
